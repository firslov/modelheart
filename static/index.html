Claude Code 配置

文件路径: ~/.claude/settings.json 

{
  "env": {
    "ANTHROPIC_AUTH_TOKEN": "your key",
    "ANTHROPIC_BASE_URL": "{{API_BASE_URL}}/anthropic",
    "ANTHROPIC_DEFAULT_HAIKU_MODEL": "kimi-claude",
    "ANTHROPIC_DEFAULT_SONNET_MODEL": "kimi-claude",
    "ANTHROPIC_DEFAULT_OPUS_MODEL": "kimi-claude"
  },
  "model": "kimi-claude",
  "outputStyle": "Default"
}

OpenCode 配置

文件路径: ~/.config/opencode/opencode.json 

{
  "$schema": "https://opencode.ai/config.json",
  "provider": {
    "HaoAI": {
      "models": {
        "haoai/kimi": {
          "name": "haoai/kimi"
        }
      },
      "npm": "@ai-sdk/anthropic",
      "options": {
        "apiKey": "your key",
        "baseURL": "{{API_BASE_URL}}/anthropic/v1"
      }
    },
    "HaoAICoding": {
      "models": {
        "haoai/glm5": {
          "name": "haoai/glm5"
        },
        "haoai/glm4.7": {
          "name": "haoai/glm4.7"
        }
      },
      "name": "HaoAI Coding",
      "npm": "@ai-sdk/openai-compatible",
      "options": {
        "apiKey": "your key",
        "baseURL": "{{API_BASE_URL}}/coding"
      }
    }
  }
}
